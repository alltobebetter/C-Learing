#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

int hourglass(int time, int downstar) {
    if (time < 0) {
       exit(0); 
    }

    int upblock = 16 - time;
    int rows[] = {7, 5, 3, 1};

    for (int i = 0; i < 4; i++) {
        for (int j = 0; j < rows[i]; j++) {
            if (upblock > 0) {
                printf("-");
                upblock--;
            } else {
                printf("*");
            }
        }
        printf("\n");
    }

    for (int i = 3; i >= 0; i--) {
        int downblock = 16 - downstar;
        for (int j = 0; j < rows[i]; j++) {
            if (downblock > 0) {
                printf("-");
                downblock--;
            } else {
                printf("*");
            }
        }
        printf("\n");
    }

    sleep(1);
    system("clear");
    hourglass(time - 1, downstar + 1); 

    return 0;
}

int main() {
    int time;
    printf("请输入你要计算的秒数，范围在1~16：");
    scanf("%d", &time);
    hourglass(time, 0);
    return 0;
}
